# FOR minimum salaries

SELECT DISTINCT SALARY FROM JOBS E1 
 WHERE 2 >= (SELECT COUNT(DISTINCT SALARY)FROM JOBS E2 
  WHERE E1.SALARY >= E2.SALARY) ORDER BY E1.SALARY DESC;

# For maximum salaries

SELECT DISTINCT SALARY FROM JOBS E1 
 WHERE 2 >= (SELECT COUNT(DISTINCT SALARY)FROM JOBS E2 
  WHERE E1.SALARY <= E2.SALARY) ORDER BY E1.SALARY DESC;
